import { Component, OnInit, Input, OnChanges, SimpleChanges } from '@angular/core';
import { SendDataService } from './../send-data.service';

@Component({
  selector: 'app-confirmpaymentmodal',
  templateUrl: './confirmpaymentmodal.component.html',
  styleUrls: ['./confirmpaymentmodal.component.sass']
})
export class ConfirmpaymentmodalComponent implements OnInit, OnChanges {

  @Input() id: any;

  @Input() selectCoachAmount = 0;
  @Input() selectedCoachId: any;

  CouponText: any = 'Coupon';
  couponcode: any = '';
  constructor(private send: SendDataService) { }
  ngOnChanges(change: SimpleChanges) {
  }
  ngOnInit() {
    window.scrollTo(0, 0);
  }
  cancelPay() {
    this.selectCoachAmount = undefined;
    this.selectedCoachId = undefined;
  }

  sendCoachData() {
    this.send.selectCoach(this.selectedCoachId, this.id);
  }

  coupon() {
    this.send.couponIn = !this.send.couponIn;
    // this.CouponText = this.send.couponIn ? 'Skip' : 'Coupon';
  }
  couponApply(){
    if (this.couponcode) {
      if (this.couponcode.trim().length) {
        this.send.coupon = this.couponcode;
        this.send.getCouponDetails(this.selectedCoachId);
      }
    }
  }
}

<div class="overlay" *ngIf="selectCoachAmount">
  <div class="confirmPayment text-center">
    <span *ngIf="!send.couponIn">
      {{selectCoachAmount}} will be deducted from your card.

    </span>
    <div *ngIf="send.couponIn"  class="coupon">
      if you have a coupon for discount, please provide the coupon code.

      </div>

        Coupon

        Skip

      <button  *ngIf="send.couponIn" (click)="couponApply()" class="btn btn-success" data-dismiss="modal">

        OK
      </button>
