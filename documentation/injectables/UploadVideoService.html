import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Http, Response, Headers, RequestOptions } from '@angular/http';
import { Router, ActivatedRoute } from '@angular/router';
import { Subject } from 'rxjs/Subject';
import { Observable } from 'rxjs/Observable';
import 'rxjs/Rx';
import * as AWS from 'aws-sdk/global';
import * as S3 from 'aws-sdk/clients/s3';
import { ToastrService } from 'ngx-toastr';
import { AuthService } from './auth.service';

@Injectable()
export class UploadVideoService {
  videoURL = new Subject();
  uploadPercent: any = '';
  showPercent: any = false;
  constructor(public http: Http,
    public router: Router,
    public route: ActivatedRoute,
  private notify: ToastrService, private auth: AuthService) {
  }

  uploadVideo(video, id?) {
    this.s3(video);
    let route = this.router.url;
    if (route.search('coachReview') > -1) {
      return true;
    }
    if (route.includes('coachProfile')) {
      this.router.navigate(['/app/createPost', id], { relativeTo: this.route });
    } else {
      this.router.navigate(['/app/createPost'], { relativeTo: this.route });
    }
  }

  getVideourl(): Observable<any> {
    return this.videoURL.asObservable();
  }

  s3(video: any): any {
    this.showPercent = true;
    const bucket = new S3(
      {
        accessKeyId: 'AKIAJ7TNC6TQAM3X4BFA',
        secretAccessKey: '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',
        region: 'us-east-1',
      }
    );
    const params = {
      Bucket: 'rtc-production',
      Key: 'attachments/videos/' + video.name,
      ACL: 'public-read',
      AccelerateConfiguration: {
        Status: 'Enabled'
      },
      Body: video
    };
    bucket.upload(params, (err, res) => {
      if (err) {
        this.notify.error('Video Upload Not Successful');
        console.log(err);
        // this.router.navigate(['/app/videos'], { relativeTo: this.route });
        return false;
      }
      this.notify.success('Video Upload Successful');
      this.videoURL.next(res.Location);
      this.showPercent = false;
      this.uploadPercent = 0;
    }).on('httpUploadProgress', (evt) => {
      this.uploadPercent = Math.round(((evt.loaded) / evt.total) * 100) ;
    });
  }

}
